578 template<class T> quaternion<T> quaternion<T>::slerp(const quaternion<T>& a, const quaternion<T>& b, double t) {
579 double dot = quaternion<T>::dot(a, b);
580 if (dot < 0) {
581 a = -a;
582 dot = -dot;
583 }
584
585 const double DOT_THRESHOLD = 0.995;
586 if (dot > DOT_THRESHOLD) {
587 quaternion<T> out = a + (b-a)*t;
588 out = out.normalize();
589 return out;
590 }
591
592 double theta_0 = acos(dot);
593 double theta = theta_0 * t;
594
595 double sin_theta = sin(theta);
596 double sin_theta_0 = sin(theta_0);
597
598 double s0 = cos(theta) - dot * sin_theta / sin_theta_0;
599 double s1 = sin_theta / sin_theta_0;
600
601 return (a*s0) + (b*s1);
602 }
